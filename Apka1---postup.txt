Logika v pseudokóde — aby si mal predstavu čo Claude Code spraví:

function search(query):
    for hours in [24, 48, 72]:
        results = findMessages(query, withinHours=hours)
        if results.length > 0:
            show("Nájdené v posledných " + hours + " hodinách")
            return results

    show("Nič sa nenašlo ani za 72 hodín")
    return []
